Python (requests) - Chat

Make a chat request using Python's requests library.

import requests

response = requests.post(
    "https://hager-zon.vercel.app/api/chat",
    json={
        "message": "Generate a React component",
        "system_prompt": "You are a coding expert",
        "temperature": 0.7,
        "max_new_tokens": 4096
    }
)
print(response.json())

Python (Django) - Chat

Integrate with a Django project.

import requests
from django.http import JsonResponse
from django.views import View

class ChatView(View):
    def post(self, request):
        data = request.POST.get('message')
        response = requests.post(
            "https://hager-zon.vercel.app/api/chat",
            json={"message": data, "system_prompt": "You are a coding expert", "temperature": 0.7}
        )
        return JsonResponse(response.json())

Python (Flask) - Chat

Integrate with a Flask app.

from flask import Flask, request, jsonify
import requests

app = Flask(__name__)

@app.route('/chat', methods=['POST'])
def chat():
    data = request.json.get('message')
    response = requests.post(
        "https://hager-zon.vercel.app/api/chat",
        json={"message": data, "system_prompt": "You are a coding expert", "temperature": 0.7}
    )
    return jsonify(response.json())

JavaScript (Node.js/Express) - Chat

Integrate with a Node.js/Express server.

const express = require('express');
const fetch = require('node-fetch');
const app = express();

app.use(express.json());

app.post('/chat', async (req, res) => {
    const { message } = req.body;
    const response = await fetch('https://hager-zon.vercel.app/api/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            message,
            system_prompt: 'You are a coding expert',
            temperature: 0.7,
            max_new_tokens: 4096
        })
    });
    const data = await response.json();
    res.json(data);
});

app.listen(3000, () => console.log('Server running on port 3000'));

Ruby (Rails) - Chat

Integrate with a Ruby on Rails app.

require 'httparty'

class ChatController < ApplicationController
    def create
        response = HTTParty.post(
            'https://hager-zon.vercel.app/api/chat',
            body: {
                message: params[:message],
                system_prompt: 'You are a coding expert',
                temperature: 0.7,
                max_new_tokens: 4096
            }.to_json,
            headers: { 'Content-Type' => 'application/json' }
        )
        render json: response.parsed_response
    end
end

PHP (Laravel) - Chat

Integrate with a Laravel app.

use Illuminate\Support\Facades\Http;

class ChatController extends Controller
{
    public function chat(Request $request)
    {
        $response = Http::post('https://hager-zon.vercel.app/api/chat', [
            'message' => $request->input('message'),
            'system_prompt' => 'You are a coding expert',
            'temperature' => 0.7,
            'max_new_tokens' => 4096
        ]);
        return $response->json();
    }
}
